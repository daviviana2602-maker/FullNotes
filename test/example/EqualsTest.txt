@Test
    void shouldCreateAccountSuccessfully() {

        when(userRepository.findByEmail("davi@gmail.com"))
                .thenReturn(Optional.empty());

        when(passwordEncoder.encode("123456"))
                .thenReturn("HASH");


        CreateAccountResponse response =
                createAccountService.createAccount(
                        "Davi",
                        "davi@gmail.com",
                        "123456"
                );


        assertEquals("Davi", response.name());
        assertEquals("davi@gmail.com", response.email());
        assertEquals(UserRoleEnum.USER, response.role());
        assertEquals(UserStatusEnum.ACTIVE, response.status());


        verify(passwordEncoder)
                .encode("123456");

        verify(userRepository)
                .saveAndFlush(any(UserEntity.class));          

    }
